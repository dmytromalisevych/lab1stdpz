 96def create_access_token(data: dict):
 97    """
 98    Генерує JWT токен для авторизації користувача.
 99    
100    Args:
101        data (dict): Дані для кодування в токен (наприклад, username).
102        
103    Returns:
104        str: Згенерований JWT токен.
105    """
106    to_encode = data.copy()
107    expire = datetime.utcnow() + timedelta(minutes=ACCESS_TOKEN_EXPIRE_MINUTES)
108    to_encode.update({"exp": expire})
109    encoded_jwt = jwt.encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)
110    return encoded_jwt

112async def get_current_user(request: Request):
113    """
114    Отримує поточного авторизованого користувача з кукі (JWT токен).
115    
116    Args:
117        request (Request): Об'єкт HTTP-запиту з кукі.
118        
119    Returns:
120        User | None: Об'єкт користувача, якщо токен валідний, інакше None.
121    """
122    token = request.cookies.get("access_token")
123    if not token:
124        return None
125    try:
126        token_type, _, token_value = token.partition(" ")
127        if token_type.lower() != "bearer":
128            return None
129        
130        payload = jwt.decode(token_value, SECRET_KEY, algorithms=[ALGORITHM])
131        username = payload.get("sub")
132        if not username:
133            return None
134            
135        db = SessionLocal()
136        user = db.query(User).filter(User.username == username).first()
137        db.close()
138        return user
139    except:
140        return None
